1. Operators

Arithmetic Operators

Used for mathematical operations.

a = 10
b = 3
print(a + b)
print(a - b)
print(a * b)
print(a / b)
print(a % b)

Relational Operators

Used to compare values (True / False).

x = 5
y = 10
print(x < y)
print(x == y)
print(x != y)

Logical Operators

a = 10
b = 5
print(a > b and b > 0)
print(a < b or b > 0)
print(not(a > b))

Assignment Operators

a = 10
a += 5
print(a)

2. Data Types

Primitive Data Types

  • Integer
  • Float
  • String
  • Boolean
a = 10
b = 2.5
c = "Python"
d = True
print(type(a), type(b), type(c), type(d))

Non-Primitive Data Types

List

nums = [1, 2, 3]
nums.append(4)
print(nums)

Tuple

t = (10, 20, 30)
print(t)

Set

s = {1, 2, 2, 3}
print(s)

Dictionary

student = {"name": "Pavan", "age": 21}
print(student["name"])

3. Conditional Statements

if

age = 18
if age >= 18:
    print("Eligible to vote")

if-else

num = 5
if num % 2 == 0:
    print("Even")
else:
    print("Odd")

if-elif-else

marks = 85
if marks >= 90:
    print("A Grade")
elif marks >= 75:
    print("B Grade")
else:
    print("C Grade")

Ternary Operator

a = 10
b = 20
print("a is big" if a > b else "b is big")

4. Loops

for Loop

for i in range(1, 6):
    print(i)

while Loop

i = 1
while i <= 5:
    print(i)
    i += 1

6. OOPS (Object Oriented Programming)

Class & Object

class Student:
    def show(self):
        print("Student class")

s = Student()
s.show()

Constructor

class Student:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def display(self):
        print(self.name, self.age)

s = Student("Pavan", 21)
s.display()

Inheritance

class Parent:
    def show(self):
        print("Parent class")

class Child(Parent):
    def display(self):
        print("Child class")

c = Child()
c.show()
c.display()

Polymorphism

class A:
    def show(self):
        print("Class A")

class B:
    def show(self):
        print("Class B")

A().show()
B().show()

Encapsulation

class Bank:
    def __init__(self, balance):
        self.__balance = balance

    def show(self):
        print(self.__balance)

b = Bank(5000)
b.show()

Abstraction

from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        pass

class Square(Shape):
    def area(self):
        print("Square Area")

Square().area()
